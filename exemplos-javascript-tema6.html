Exemplos JavaScript - Tema 6

Os eventos da linguaxe de guión: formulario, rato, teclado, foco, xanela e boas prácticas con addEventListener().

Definición de eventos

Un evento executa código cando o usuario realiza unha acción na páxina.

boton.addEventListener("click", function () {
  console.log("Evento click executado");
});

Accións asociadas aos eventos

let boton = document.getElementById("btn");

boton.addEventListener("click", mostrar);

function mostrar() {
  console.log("Fixeches clic");
}

Xerarquía dos eventos desde Window

Obxecto Función
window Xanela do navegador
document Documento HTML cargado
Elementos HTML Botóns, formularios, textos, imaxes... 
window.addEventListener("load", iniciar);

function iniciar() {
  console.log("Páxina cargada");
}

Evento onselect

input.addEventListener("select", seleccionar);

function seleccionar() {
  console.log("Texto seleccionado");
}

Evento onchange

cidade.addEventListener("change", cambiar);

function cambiar() {
  console.log(cidade.value);
}

Eventos de rato

Evento Descrición
mousedown Pulsar un botón do rato
mousemove Mover o rato
mouseout Saír dun elemento
mouseover Entrar nun elemento
mouseup Soltar un botón do rato
click Facer clic 
elemento.addEventListener("mouseover", entrar);
elemento.addEventListener("mousemove", mover);
elemento.addEventListener("mouseout", sair);
Move o rato por esta zona e fai clic.

Eventos de teclado

Evento Descrición
keydown Pulsar unha tecla
keypress Manter unha tecla pulsada
keyup Soltar unha tecla 
campo.addEventListener("keyup", escribir);

function escribir() {
  console.log(campo.value);
}

Eventos de enfoque

input.addEventListener("focus", activar);
input.addEventListener("blur", validar);

Eventos de formulario

formulario.addEventListener("submit", enviar);
formulario.addEventListener("reset", limpar);

function enviar(event) {
  event.preventDefault();
  console.log("Formulario enviado");
}

Eventos de xanela

Evento Descrición
load Carga completa da páxina
resize Cambio de tamaño da xanela
unload Saída da páxina
move Movemento da xanela, con pouco uso actual 
window.addEventListener("resize", cambiarTamano);

function cambiarTamano() {
  console.log(window.innerWidth);
}

Boas prácticas actuais

Recoméndase utilizar addEventListener() para separar o HTML do JavaScript e permitir varios eventos nun mesmo elemento.

boton.addEventListener("click", mostrar);

Manipulación habitual do DOM con eventos

Propiedade ou método Descrición
textContent Cambia texto dun elemento
innerHTML Insire HTML dentro dun elemento
value Obtén o valor dun input
classList.toggle() Activa ou desactiva unha clase
style Modifica estilos CSS 
menu.classList.toggle("visible");
Texto inicial

Exercicio práctico

Crear un botón de menú móbil que mostra e oculta a navegación usando click, querySelector() e classList.toggle().

let boton = document.getElementById("btnMenu");
let menu = document.querySelector(".nav-mobile");

boton.addEventListener("click", mostrar);

function mostrar() {
  menu.classList.toggle("visible");
}
