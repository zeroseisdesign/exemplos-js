Exercicio 1: Formulario de Rexistro

Implementación paso a paso dun formulario de rexistro con validacións.

Requisitos:

  • Campos: Nome, Apelidos, Email, Idade, Contrasinal
  • Todos obrigatorios
  • Idade entre 18 e 99
  • Contrasinal mínimo 8 caracteres
  • Feedback visual con CSS

Paso 1: Estrutura HTML do formulario

Definimos o formulario cos campos necesarios. Cada campo leva o seu label, type axeitado e atributos de validación HTML5.

<form id="form-rexistro" action="#" method="get">
    <fieldset>
        <legend>Datos de rexistro</legend>

        <label for="rex-nome">Nome:</label>
        <input type="text" id="rex-nome" required>
        <div class="mensaxe-erro" id="erro-nome"></div>

        <label for="rex-apelidos">Apelidos:</label>
        <input type="text" id="rex-apelidos" required>
        <div class="mensaxe-erro" id="erro-apelidos"></div>

        <label for="rex-email">Email:</label>
        <input type="email" id="rex-email" required>
        <div class="mensaxe-erro" id="erro-email"></div>

        <label for="rex-idade">Idade:</label>
        <input type="number" id="rex-idade" min="18" max="99" required>
        <div class="mensaxe-erro" id="erro-idade"></div>

        <label for="rex-password">Contrasinal:</label>
        <input type="password" id="rex-password" minlength="8" required>
        <div class="mensaxe-erro" id="erro-password"></div>

        <br>
        <button type="submit">Rexistrarse</button>
        <button type="reset">Limpar</button>
    </fieldset>
</form>

Paso 2: Estilos CSS para feedback visual

Usamos as pseudoclases :valid, :invalid e :focus:invalid para dar feedback visual automático sen JavaScript.

/* Campos válidos - bordo verde */
input:valid {
    border: 2px solid green;
}

/* Campos inválidos - bordo vermello */
input:invalid {
    border: 2px solid red;
}

/* Campo inválido con foco - fondo vermello claro */
input:focus:invalid {
    background-color: #ffe5e5;
}

/* Mensaxes de erro */
.mensaxe-erro {
    color: red;
    font-size: 0.9em;
    margin-top: 4px;
}

.mensaxe-valida {
    color: green;
    font-size: 0.9em;
    margin-top: 4px;
}

Paso 3: Funcións de validación individuais

Creamos unha función para cada campo. Isto organiza mellor o código e permite reutilizar as validacións.

function validarCampoTexto(id, min, max) {
    let campo = document.getElementById(id);
    let valor = campo.value.trim();
    if (valor === "") {
        return "Campo obrigatorio";
    }
    if (valor.length < min) {
        return "Mínimo " + min + " caracteres";
    }
    if (max && valor.length > max) {
        return "Máximo " + max + " caracteres";
    }
    return "";
}

function validarEmail(id) {
    let campo = document.getElementById(id);
    let valor = campo.value.trim();
    if (valor === "") {
        return "Campo obrigatorio";
    }
    if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(valor)) {
        return "Formato de email incorrecto";
    }
    return "";
}

function validarIdade(id, min, max) {
    let campo = document.getElementById(id);
    let valor = parseInt(campo.value, 10);
    if (isNaN(valor)) {
        return "Campo obrigatorio";
    }
    if (valor < min) {
        return "A idade mínima é " + min;
    }
    if (valor > max) {
        return "A idade máxima é " + max;
    }
    return "";
}

function validarPassword(id, min) {
    let campo = document.getElementById(id);
    let valor = campo.value;
    if (valor === "") {
        return "Campo obrigatorio";
    }
    if (valor.length < min) {
        return "Mínimo " + min + " caracteres (actuais: " + valor.length + ")";
    }
    return "";
}
Nota: Cada función devolve unha cadea baleira se é válido, ou unha mensaxe de erro se non é válido.

Paso 4: Validación en tempo real con evento input

Asociamos o evento input a cada campo para validar mentres o usuario escribe. Así o usuario ve o feedback inmediatamente.

function mostrarMensaxe(idErro, mensaxe) {
    let div = document.getElementById(idErro);
    if (mensaxe) {
        div.className = "mensaxe-erro";
        div.textContent = mensaxe;
    } else {
        div.className = "mensaxe-valida";
        div.textContent = "✓ Correcto";
    }
}

// Validación en tempo real para cada campo
document.getElementById("rex-nome").addEventListener("input", function () {
    let erro = validarCampoTexto("rex-nome", 2, 50);
    mostrarMensaxe("erro-nome", erro);
});

document.getElementById("rex-apelidos").addEventListener("input", function () {
    let erro = validarCampoTexto("rex-apelidos", 2, 100);
    mostrarMensaxe("erro-apelidos", erro);
});

document.getElementById("rex-email").addEventListener("input", function () {
    let erro = validarEmail("rex-email");
    mostrarMensaxe("erro-email", erro);
});

document.getElementById("rex-idade").addEventListener("input", function () {
    let erro = validarIdade("rex-idade", 18, 99);
    mostrarMensaxe("erro-idade", erro);
});

document.getElementById("rex-password").addEventListener("input", function () {
    let erro = validarPassword("rex-password", 8);
    mostrarMensaxe("erro-password", erro);
});

Paso 5: Validación ó enviar o formulario

Cando o usuario fai clic en "Rexistrarse", validamos todos os campos. Se hai algún erro, cancelamos o envío con preventDefault() e mostramos unha alerta.

document.getElementById("form-rexistro").addEventListener("submit", function (event) {
    let erros = [];

    let erroNome = validarCampoTexto("rex-nome", 2, 50);
    if (erroNome) erros.push("Nome: " + erroNome);

    let erroApelidos = validarCampoTexto("rex-apelidos", 2, 100);
    if (erroApelidos) erros.push("Apelidos: " + erroApelidos);

    let erroEmail = validarEmail("rex-email");
    if (erroEmail) erros.push("Email: " + erroEmail);

    let erroIdade = validarIdade("rex-idade", 18, 99);
    if (erroIdade) erros.push("Idade: " + erroIdade);

    let erroPassword = validarPassword("rex-password", 8);
    if (erroPassword) erros.push("Contrasinal: " + erroPassword);

    if (erros.length > 0) {
        event.preventDefault();
        alert("Erros no formulario:\n- " + erros.join("\n- "));
    } else {
        event.preventDefault();
        let output = document.querySelector("#demo-final .output");
        output.textContent = "✓ Formulario enviado correctamente!\n";
        output.textContent += "   Nome: " + document.getElementById("rex-nome").value + "\n";
        output.textContent += "   Apelidos: " + document.getElementById("rex-apelidos").value + "\n";
        output.textContent += "   Email: " + document.getElementById("rex-email").value + "\n";
        output.textContent += "   Idade: " + document.getElementById("rex-idade").value + "\n";
        output.textContent += "   Contrasinal: " + "*".repeat(document.getElementById("rex-password").value.length);
    }
});
