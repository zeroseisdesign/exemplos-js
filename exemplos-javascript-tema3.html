Sentenza IF

if (idade >= 18) {
  console.log("Maior de idade");
}

Sentenza WHILE

let contador = 1;

while (contador <= 5) {
  console.log(contador);
  contador++;
}

Sentenza FOR

for (let i = 1; i <= 5; i++) {
  console.log(i);
}

Sentenza BREAK

for (let i = 1; i <= 10; i++) {

  if (i === 5) {
    break;
  }

  console.log(i);
}

Sentenza CONTINUE

for (let i = 1; i <= 5; i++) {

  if (i === 3) {
    continue;
  }

  console.log(i);
}

Sentenza SWITCH

let dia = 2;

switch (dia) {

  case 1:
    console.log("Luns");
    break;

  case 2:
    console.log("Martes");
    break;

  default:
    console.log("Outro día");
}
