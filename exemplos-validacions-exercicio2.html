Exercicio 2: Validación en Tempo Real

Implementación paso a paso dun sistema de validación que reacciona mentres o usuario escribe.

Requisitos:

  • O usuario escribe e o campo cambia de cor automaticamente
  • Mostrar mensaxe baixo o campo indicando o estado
  • Feedback visual inmediato sen necesidade de premer ningún botón

Paso 1: Estrutura HTML do campo

Creamos un campo de texto simple cun div para a mensaxe de feedback. Non leva button porque a validación é automática.

<label for="campo-tempo-real">Nome de usuario:</label>
<input type="text" id="campo-tempo-real" placeholder="Escribe aquí...">
<div id="feedback-tempo-real" class="mensaxe-info">
    Agardando entrada...
</div>

Paso 2: Estilos CSS para os estados

Definimos tres estilos visuais: estado neutro (sen escribir), erro (inválido) e válido. A transición suave fai que o cambio sexa agradable.

Paso 3: Función de validación con criterios

A función recibe o valor e devolve obxectos co resultado e a mensaxe. Así podemos comprobar múltiples condicións.

function validarUsuario(valor) {
    let resultado = {
        valido: false,
        mensaxe: ""
    };

    if (valor === "") {
        resultado.mensaxe = "Escribe algo...";
        return resultado;
    }

    if (valor.length < 3) {
        resultado.mensaxe = "Necesítanse polo menos 3 caracteres (levas " + valor.length + ")";
        return resultado;
    }

    if (valor.length > 20) {
        resultado.mensaxe = "Máximo 20 caracteres (" + valor.length + " exceden o límite)";
        return resultado;
    }

    resultado.valido = true;
    resultado.mensaxe = "✓ \"" + valor + "\" é correcto (" + valor.length + " caracteres)";
    return resultado;
}
Nota: Usar obxectos como valor de retorno é unha boa práctica porque permite devolver múltiples datos (válido + mensaxe) nun só lugar.

Paso 4: Evento input con cambio de cor e mensaxe

O evento input disparase cada vez que o campo cambia. Aquí actualizamos a clase CSS e a mensaxe baixo o campo.

document.getElementById("campo-tempo-real")
    .addEventListener("input", function () {

    let feedback = document.getElementById("feedback-tempo-real");
    let resultado = validarUsuario(this.value);

    // Cambiar clase CSS do campo
    if (this.value === "") {
        this.className = "neutro";
        feedback.className = "mensaxe-info";
    } else if (resultado.valido) {
        this.className = "valido";
        feedback.className = "mensaxe-valida";
    } else {
        this.className = "erro";
        feedback.className = "mensaxe-erro";
    }

    feedback.textContent = resultado.mensaxe;
});

Paso 5: Engadir barra de progreso visual

Podemos mellorar o feedback visual cunha barra de progreso que mostre cantos caracteres leva o usuario respecto ao mínimo.

function actualizarBarra(valor, min, max) {
    let barra = document.querySelector(".barra-progreso .cheo");
    let porcentaxe;

    if (valor === 0) {
        porcentaxe = 0;
    } else if (valor < min) {
        porcentaxe = (valor / min) * 50; // 0-50%
    } else if (valor <= max) {
        porcentaxe = 50 + ((valor - min) / (max - min)) * 50; // 50-100%
    } else {
        porcentaxe = 100;
    }

    barra.style.width = Math.min(porcentaxe, 100) + "%";
}

Paso 6: Múltiples campos con validación en tempo real

Podemos aplicar o mesmo patrón a varios campos. Cada campo ten o seu propio feedback e barra de progreso.

function configurarCampo(idCampo, idFeedback, idBarra, min, max, nome) {
    let campo = document.getElementById(idCampo);
    let feedback = document.getElementById(idFeedback);
    let barra = document.getElementById(idBarra);

    campo.addEventListener("input", function () {
        let valor = this.value;
        let longitud = valor.length;

        // Validar
        let valido = longitud >= min && longitud <= max;
        let mensaxe = "";

        if (valor === "") {
            mensaxe = "Escribe o " + nome + "...";
            campo.className = "neutro";
            feedback.className = "mensaxe-info";
        } else if (longitud < min) {
            mensaxe = "Faltan " + (min - longitud) + " caracteres para o mínimo";
            campo.className = "erro";
            feedback.className = "mensaxe-erro";
        } else if (longitud > max) {
            mensaxe = "Excedes o máximo en " + (longitud - max) + " caracteres";
            campo.className = "erro";
            feedback.className = "mensaxe-erro";
        } else {
            mensaxe = "✓ " + nome + " correcto (" + longitud + " caracteres)";
            campo.className = "valido";
            feedback.className = "mensaxe-valida";
        }

        feedback.textContent = mensaxe;
        if (barra) {
            actualizarBarra(longitud, min, max, barra);
        }
    });
}

// Configurar varios campos
configurarCampo("campo-nome", "fb-nome", "barra-nome", 3, 15, "nome");
configurarCampo("campo-usuario", "fb-usuario", "barra-usuario", 4, 12, "usuario");
configurarCampo("campo-contrasinal", "fb-contrasinal", null, 6, 20, "contrasinal");
Nota: Esta función configurarCampo recibe os IDs dos elementos e os límites. Isto fai que o código sexa reutilizable e fácil de modificar.

Demo 3: Indicador de forza do contrasinal

Un indicador visual que mostra a seguridade do contrasinal baseado en múltiples criterios.

function avaliarForza(password) {
    let puntos = 0;
    if (password.length >= 8) puntos++;
    if (password.length >= 12) puntos++;
    if (/[a-z]/.test(password) && /[A-Z]/.test(password)) puntos++;
    if (/[0-9]/.test(password)) puntos++;
    if (/[^a-zA-Z0-9]/.test(password)) puntos++;
    return puntos;
}
